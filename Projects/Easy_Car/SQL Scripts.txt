DROP TABLE TEAM1.CAR;
DROP TABLE TEAM1.customer_information;
DROP TABLE TEAM1.Light_PR;
DROP TABLE TEAM1.Bumper_PR;
DROP TABLE TEAM1.Tire_PR;

CREATE TABLE Car(
car_code VARCHAR2(20) NOT NULL PRIMARY KEY,
light VARCHAR2(10) NOT NULL,
bumper VARCHAR2(10) NOT NULL,
tr_size NUMBER NOT NULL);

CREATE TABLE Customer_information(
customer_no VARCHAR2(20) NOT NULL PRIMARY KEY);

CREATE TABLE Light_PR(
lt_code VARCHAR2(20) NOT NULL PRIMARY KEY,
lt_price NUMBER NOT NULL);
 
CREATE TABLE Bumper_PR(
bp_code VARCHAR2(20) NOT NULL PRIMARY KEY,
bp_price NUMBER NOT NULL);

CREATE TABLE Tire_PR(
tr_code VARCHAR2(20) NOT NULL PRIMARY KEY,
tr_size NUMBER NOT NULL,
tr_price NUMBER NOT NULL,
br_name VARCHAR2(20) NOT NULL);

ALTER TABLE customer_information
ADD FOREIGN KEY (Car_Code)
REFERENCES car(Car_Code);
